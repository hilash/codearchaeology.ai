	private void createMenuOverlays()
	{
		// TODO - find more elegant solution
        // assign Pop up item with overlay's
		overlaysMenuItemsChecks.put(R.id.menu_overlay_angels, true);
		overlaysMenuItemsChecks.put(R.id.menu_overlay_info, true);
		overlaysMenuItemsChecks.put(R.id.menu_overlay_personal, false);
		overlaysMenuItemsChecks.put(R.id.menu_overlay_travelers, true);
		overlaysMenuItemsChecks.put(R.id.menu_overlay_water, true);
		overlaysMenuItemsChecks.put(R.id.menu_overlay_water_hide, true);
		
		for (Integer id : overlaysMenuItemsChecks.keySet())
		{
            // add overlay if checked
            if (overlaysMenuItemsChecks.get(id)){
                Overlay overlay = getOverlayByID(id);
                mapView.getOverlays().add(overlay);
            	MenuItemsIdOverlay.put(id, (ArrayItemizedOverlay) overlay);
            }
        }
	}
	
    public void showPopupOverlay(View v) {
        PopupMenu popup = new PopupMenu(this, v);
        MenuInflater inflater = popup.getMenuInflater();
        inflater.inflate(R.menu.map_overlay_menu, popup.getMenu());
        
        // loop for menu items
        for (int i = 0; i < popup.getMenu().size(); ++i)
        {
            MenuItem item = popup.getMenu().getItem(i);
            int id = item.getItemId();
            if (!overlaysMenuItemsChecks.containsKey(id))
            {
            	overlaysMenuItemsChecks.put(id, item.isChecked());
            }
            boolean check = overlaysMenuItemsChecks.get(id);
            item.setChecked(check); 
        }
        
        popup.setOnMenuItemClickListener(new PopupMenu.OnMenuItemClickListener() {
            public boolean onMenuItemClick(MenuItem item) {
            	int id = item.getItemId();

                if (item.isChecked()){               	
                	Overlay overlay = MenuItemsIdOverlay.get(id);
                    mapView.getOverlays().remove(overlay);
                    MenuItemsIdOverlay.remove(id);
                }
                else {
                	// add it
                    ArrayItemizedOverlay overlay = getOverlayByID(id);
                    mapView.getOverlays().add(overlay);
                	MenuItemsIdOverlay.put(id, overlay);
                }
                
                // common
                overlaysMenuItemsChecks.put(id, !item.isChecked());
                item.setChecked(!item.isChecked());
                return true;
            }
        });
        popup.show();
    }

	public void onReportDialogClick(View v) {
		GeneralItemType type;
		switch(v.getId()){
			case R.id.report_dialog_button_angel:
				Intent intent = new Intent(myApplication, AngelEditor.class);
				intent.setAction(Intent.ACTION_INSERT);
				intent.putExtra(AngelEditor.INTENT_IN_DATA_ANGEL_LAT, LongPressGeoPoint.getLatitude());
				intent.putExtra(AngelEditor.INTENT_IN_DATA_ANGEL_LON, LongPressGeoPoint.getLongitude());
				intent.putExtra(AngelEditor.INTENT_IN_DATA_ANGEL_POSITION, 0);
				mainActivity.startActivityForResult(intent, INTENT_REQ_CODE_ANGEL_INSERT);
				break;
			case R.id.report_dialog_button_info:
				type = GeneralItemType.GENERAL_ITEM_TYPE_INFORMATION;
			case R.id.report_dialog_button_water:
				type = GeneralItemType.GENERAL_ITEM_TYPE_WATER;
			case R.id.report_dialog_button_water_hide:
				type = GeneralItemType.GENERAL_ITEM_TYPE_WATER_HIDE;
				Intent intent1 = new Intent(myApplication, GeneralItemEditor.class);
				intent1.setAction(Intent.ACTION_INSERT);
				intent1.putExtra(GeneralItemEditor.INTENT_IN_DATA_GENERAL_TYPE, type);
				intent1.putExtra(GeneralItemEditor.INTENT_IN_DATA_GENERAL_LAT, LongPressGeoPoint.getLatitude());
				intent1.putExtra(GeneralItemEditor.INTENT_IN_DATA_GENERAL_LON, LongPressGeoPoint.getLongitude());
				mainActivity.startActivityForResult(intent1, INTENT_REQ_CODE_GENERAL_INSERT);
			default:
				break;
		}
		report_dialog.dismiss();	
	}

	private void onActivityResultGeneralItem(int requestCode, int resultCode, Intent data)
    {
		// can return from edit note or inserting a new one
		if ((resultCode != Activity.RESULT_OK) || (data == null)){
			return;
		}
		
		Long itemID = data.getLongExtra(GeneralItemEditor.INTENT_OUT_DATA_GENERAL_ID, Long.MAX_VALUE);
		byte itemType = data.getByteExtra(GeneralItemEditor.INTENT_OUT_DATA_GENERAL_TYPE, Byte.MAX_VALUE);
	    double mItemLatitude = data.getDoubleExtra(GeneralItemEditor.INTENT_IN_DATA_GENERAL_LAT, Double.MAX_VALUE);
	    double mItemLongitude = data.getDoubleExtra(GeneralItemEditor.INTENT_IN_DATA_GENERAL_LON, Double.MAX_VALUE);
	    
		if (itemID == Long.MAX_VALUE){
			return;
		}
		if (requestCode == INTENT_REQ_CODE_GENERAL_INSERT){
			GeoPoint geoPoint = new GeoPoint(mItemLatitude, mItemLongitude);
			getOverlayByGeneralItemType(GeneralItemType.values()[itemType]).addItem(new OverlayItem(geoPoint, itemID.toString(), ""));
		}
    }
	
	private ArrayItemizedOverlay getOverlayByGeneralItemType(GeneralItemType type)
	{
		switch(type){
		case GENERAL_ITEM_TYPE_ERROR:
			return errorOverlay;
		case GENERAL_ITEM_TYPE_INFORMATION:
			return MenuItemsIdOverlay.get(R.id.menu_overlay_info);
		case GENERAL_ITEM_TYPE_PEOPLE:
			return MenuItemsIdOverlay.get(R.id.menu_overlay_travelers);
		//case GENERAL_ITEM_TYPE_SLEEP:
		//	return MenuItemsIdOverlay.get(R.id.menu);
		case GENERAL_ITEM_TYPE_WATER:
			return MenuItemsIdOverlay.get(R.id.menu_overlay_water);
		case GENERAL_ITEM_TYPE_WATER_HIDE:
			return MenuItemsIdOverlay.get(R.id.menu_overlay_water_hide);
		}
		return null;
	}
	
	@Override
	public void onActivityResult(int requestCode, int resultCode, Intent data)
	{
		// if overlay not exists, create it and show it (check = true)
		// if exists, add and need to refresh (somehow..)
		switch(requestCode){
		case INTENT_REQ_CODE_ANGEL_EDIT:
			onActivityResultAngel(requestCode, resultCode, data);
			break;
		case INTENT_REQ_CODE_ANGEL_INSERT:
			onActivityResultAngel(requestCode, resultCode, data);
			break;
		case INTENT_REQ_CODE_ANGEL_VIEW:
			onActivityResultAngel(requestCode, resultCode, data);
			break;
		case INTENT_REQ_CODE_GENERAL_EDIT:
			onActivityResultGeneralItem(requestCode, resultCode, data);
			break;
		case INTENT_REQ_CODE_GENERAL_INSERT:
			onActivityResultGeneralItem(requestCode, resultCode, data);
			break;
		default:
			return;
		}
	}

	private Map<GeneralItemType, Drawable> createDrawableOfOverlay()
	{
		Map<GeneralItemType, Drawable> map = new Hashtable<GeneralItemType, Drawable>();
		Resources res = myApplication.getResources();
		
		map.put(GeneralItemType.GENERAL_ITEM_TYPE_ERROR, res.getDrawable(R.drawable.ic_map_warning_48));
		map.put(GeneralItemType.GENERAL_ITEM_TYPE_INFORMATION, res.getDrawable(R.drawable.ic_map_marker_48));
		map.put(GeneralItemType.GENERAL_ITEM_TYPE_PEOPLE, res.getDrawable(R.drawable.ic_map_people_sun_32));
		map.put(GeneralItemType.GENERAL_ITEM_TYPE_SLEEP, res.getDrawable(R.drawable.ic_bed));
		map.put(GeneralItemType.GENERAL_ITEM_TYPE_WATER, res.getDrawable(R.drawable.ic_water));
		map.put(GeneralItemType.GENERAL_ITEM_TYPE_WATER_HIDE, res.getDrawable(R.drawable.ic_water_carrier));
		return map;
	}
	
	public GeneralMapOverlay getGeneralOverlay(GeneralItemType type)
	{
		
	    Drawable defaultMarker = DrawableOfOverlay.get(type);
	    GeneralMapOverlay generalMapOverlay = new GeneralMapOverlay(defaultMarker, true, this);
	    GeneralItemDao dao =  myApplication.daoSession.getGeneralItemDao();
	    List<GeneralItem> items = dao.queryBuilder().where(Properties.Type.eq(((byte)type.ordinal()))).list();
	    List<OverlayItem> overlayItems = new ArrayList<OverlayItem>(items.size());
	    for (GeneralItem item : items){
			double latitude = item.getLatitude();
			double longitude = item.getLongitude();
			GeoPoint geoPoint = new GeoPoint(latitude,longitude);
			overlayItems.add(new OverlayItem(geoPoint, item.getTitle(), item.getText()));
		}
	    items.clear();
	    generalMapOverlay.addItems(overlayItems);
	    return generalMapOverlay;        
	}
	
	GeneralMapOverlay getErrorOverlay(){
		return getGeneralOverlay(GeneralItemType.GENERAL_ITEM_TYPE_ERROR);
	}
	
	GeneralMapOverlay getInfoOverlay(){
		return getGeneralOverlay(GeneralItemType.GENERAL_ITEM_TYPE_INFORMATION);
	}
	
	GeneralMapOverlay getPeopleOverlay(){
		return getGeneralOverlay(GeneralItemType.GENERAL_ITEM_TYPE_PEOPLE);
	}
	
	GeneralMapOverlay getWaterOverlay(){
		return getGeneralOverlay(GeneralItemType.GENERAL_ITEM_TYPE_WATER);
	}
	
	GeneralMapOverlay getWaterHideOverlay(){
		return getGeneralOverlay(GeneralItemType.GENERAL_ITEM_TYPE_WATER_HIDE);
	}
